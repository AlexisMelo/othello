\begin{algorithme}
	\small
	\fonction
	{plateauRempli}
	{plateauDeJeu : Plateau}
	{\booleen}
	{i : 1 .. Longueur
	 j : 1 .. Largeur
	 estRempli : \booleen}
	{
	\affecter{i}{1}
	\affecter{j}{1}
	\affecter{estRempli}{Vrai}
	\tantque{(i <= Longueur et estRempli)}{
		\tantque{(j <= Largeur et estRempli)}{
			\affecter{estRempli}{non(caseVide(plateauDeJeu,Position(i,j)))}
			\affecter{j}{j + 1}
		}
		\affecter{i}{i + 1}
	}
	\retourner{estRempli}
	}
\end{algorithme}

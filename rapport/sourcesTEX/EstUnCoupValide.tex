\begin{algorithme}
	\small
	\fonction
	{EstCoupValide}
	{plateau : Plateau, coup : Coup}
	{\booleen}
	{}
	{
	\sialorssinon {(non EstPositionValide(plateau, ObtenirPosition(coup)))}{
        \retourner{Faux}
	}{
		\sialorssinon {(non EstPositionLibre(plateau, ObtenirPosition(coup)))}{
        	\retourner{Faux}
		}{
			\retourner{CoupPossibleDansUneDirectionQuelconque(plateau, coup)}
		}
	}
	}
\end{algorithme}

\begin{algorithme}
	\small
	\procedure
	{FaireUnePartie}
	{couleur = Couleur, ProfondeurIA = Naturel}
	{plateau = Plateau}
	{
	\affecter{plateau}{InitialiserPlateau()}
	\tantque{non FinirLaPartie(couleur, plateau}{
		\affecter{plateau}{JouerUnTour(plateau, niveau, couleur)}
		\instruction afficherPlateau(plateau)
		\affecter{couleur}{Inverser(couleur)}
	}
	}
\end{algorithme}

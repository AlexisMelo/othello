\begin{algorithme}
	\small
	\procedure
	{InitialiserPlateau}
	{\paramEntreeSortie plateau : Plateau}
	{blanc, noir, neutre : Couleur}
	{
	\affecter{neutre}{ObtenirCouleurNeutre()}
	\affecter{blanc}{ObtenirCouleurBlanc()}
	\affecter{noir}{ObtenirCouleurNoir()}
	\pour{i}{1}{ObtenirTaille(plateau)}{}{
		\pour{j}{1}{ObtenirTaille(plateau)}{}{
			\instruction{JouerCoup(plateau, CreerCoup(CreerPosition(ObtenirLigneDepuisInt(i), ObtenirColonneDepuisInt(j)), \\ \dots \dots neutre))}
		}
	}
	\instruction{JouerCoup(plateau, CreerCoup(CreerPosition(ObtenirLigneDepuisInt(ObtenirTaille(plateau)/2), \\ \dots \dots ObtenirColonneDepuisInt(ObtenirTaille(plateau)/2)), blanc))}
	\instruction{JouerCoup(plateau, CreerCoup(CreerPosition(ObtenirLigneDepuisInt(ObtenirTaille(plateau)/2 + 1), \\ \dots \dots ObtenirColonneDepuisInt(ObtenirTaille(plateau)/2 + 1)), blanc))}
	\instruction{JouerCoup(plateau, CreerCoup(CreerPosition(ObtenirLigneDepuisInt(ObtenirTaille(plateau)/2), \\ \dots \dots ObtenirColonneDepuisInt(ObtenirTaille(plateau)/2 + 1)), noir))}
	\instruction{JouerCoup(plateau, CreerCoup(CreerPosition(ObtenirLigneDepuisInt(ObtenirTaille(plateau)/2 + 1), \\ \dots \dots ObtenirColonneDepuisInt(ObtenirTaille(plateau)/2)), noir))}
	}
\end{algorithme}

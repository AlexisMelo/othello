\begin{algorithme}
	\small
	\procedure
	{Initialiser Plateau}
	{\paramEntreeSortie Plateau}
	{Taille : \naturelNonNul \\
	blanc, noir, neutre : Couleur}
	{
	\affecter{neutre}{creerNeutre()}
	\affecter{blanc}{creerBlanc()}
	\affecter{noir}{creerNoir()}
	\affecter{taille}{8}
	\pour{i}{1}{taille}{}{

		\pour{j}{1}{taille}{}{

			\instruction{JouerCoup(plateau,Coup(neutre,Position(Ligne(i),Colonne(j))))}
		}
	}

	\instruction{JouerCoup(plateau,Coup(blanc,Position(Ligne(taille div 2),Colonne(taille div 2))))}

	\instruction{JouerCoup(plateau,Coup(blanc,Position(Ligne(taille div 2 + 1),Colonne(taille div 2 + 1))))}

	\instruction{JouerCoup(plateau,Coup(noir,Position(Ligne(taille div 2),Colonne(taille div 2 + 1))))}

	\instruction{JouerCoup(plateau,Coup(noir,Position(Ligne(taille div 2 + 1),Colonne(taille div 2))))}
	}
\end{algorithme}

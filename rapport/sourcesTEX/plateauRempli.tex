\begin{algorithme}
	\small
	\fonction
	{plateauRempli}
	{plateauDeJeu : Plateau}
	{\booleen}
	{i : 1 .. Taille \\
	 j : 1 .. Taille \\
	 estRempli : \booleen}
	{
	\affecter{i}{1}
	\affecter{j}{1}
	\affecter{estRempli}{Vrai}
	\tantque{(i $\leq$ Taille et estRempli)}{
		\tantque{(j $\leq$ Taille et estRempli)}{
			\affecter{estRempli}{non EstPositionVide(plateauDeJeu,Position(i,j))}
			\affecter{j}{j + 1}
		}
		\affecter{i}{i + 1}
	}
	\retourner{estRempli}
	}
\end{algorithme}
